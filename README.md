# INSTALL


     $ pkg update && pkg upgrade
     $ termux-setup-storage
     $ pkg install python 
     $ pkg install git
     $ pkg install pip
     $ pip install stdiomask
     $ pip install simplejson
     $ pip install requests
     $ pip install futures
     $ pip install rich
     $ pip install bs4
     $ pip install mechanize
     $ rm -rf SMBF
